# Internationalization Architecture

Locale-based routing with lazy-loaded translations.

---

## Strategy

**sveltekit-i18n** for per-language lazy loading that scales to many languages.

| Aspect | Approach |
|--------|----------|
| Library | sveltekit-i18n |
| Routing | URL prefix (`/en`, `/de`) |
| Translations | JSON files, lazy-loaded per locale |
| Date/Number | Native `Intl` API |
| Detection | URL → Cookie → Accept-Language |

### Why sveltekit-i18n

| Feature | sveltekit-i18n | Paraglide |
|---------|----------------|-----------|
| Per-language loading | Yes | No (bundles all) |
| 10+ languages | ~1 KB/page | ~12+ KB/page |
| Type safety | Partial | Full |
| Runtime switching | Yes | Requires reload |
| Bundle scaling | Linear | Multiplies by languages |

For 2-5 languages, Paraglide's type safety wins. For 10+ languages, sveltekit-i18n's lazy loading is essential.

## Routing

### Layout Setup

```typescript
// src/routes/[[lang]]/+layout.ts
import { loadTranslations, locale, setRoute } from '$lib/i18n';
import type { LayoutLoad } from './$types';

export const load: LayoutLoad = async ({ url, params }) => {
  const { lang } = params;
  const { pathname } = url;

  // Set locale from URL param or default
  const currentLocale = lang || 'en';

  // Load translations for this route
  await loadTranslations(currentLocale, pathname);

  return {
    locale: currentLocale,
  };
};
```

### Route Structure

```
src/routes/
├── [[lang]]/                    # Optional locale prefix
│   ├── +layout.ts               # Load translations
│   ├── +layout.svelte
│   ├── +page.svelte             # Homepage
│   ├── about/
│   │   └── +page.svelte
│   └── dashboard/
│       └── +page.svelte
```

The `[[lang]]` optional parameter means:
- `/` → English (default)
- `/de` → German
- `/de/about` → German about page

## Usage

### In Components

```svelte
<script>
  import { t } from '$lib/i18n';
</script>

<h1>{$t('common.greeting', { name: 'World' })}</h1>

<nav>
  <a href="/">{$t('common.nav.home')}</a>
  <a href="/about">{$t('common.nav.about')}</a>
</nav>

<p>{$t('common.items', { count: 5 })}</p>
```

### Pluralization

```json
{
  "items": "{{count}} {{count; 0:items; 1:item; default:items}}",
  "messages": "You have {{count; 0:no messages; 1:one message; default:{{count}} messages}}"
}
```

```svelte
{$t('common.items', { count: 0 })}  <!-- "0 items" -->
{$t('common.items', { count: 1 })}  <!-- "1 item" -->
{$t('common.items', { count: 5 })}  <!-- "5 items" -->
```

### In Server Code

```typescript
// +page.server.ts
import { t, loadTranslations } from '$lib/i18n';

export async function load({ params }) {
  const locale = params.lang || 'en';
  await loadTranslations(locale, '/');

  return {
    title: t.get('home.title'),
    description: t.get('home.description'),
  };
}
```

## Language Switcher

### Basic Switcher

```svelte
<!-- LanguageSwitcher.svelte -->
<script>
  import { page } from '$app/stores';
  import { locale, locales } from '$lib/i18n';

  const languages = {
    en: 'English',
    de: 'Deutsch',
    fr: 'Français',
  };

  function getLocalizedPath(targetLocale: string) {
    const { pathname } = $page.url;
    // Remove current locale prefix if present
    const pathWithoutLocale = pathname.replace(/^\/(en|de|fr)/, '') || '/';

    if (targetLocale === 'en') {
      return pathWithoutLocale;
    }
    return `/${targetLocale}${pathWithoutLocale}`;
  }
</script>

<nav aria-label="Language">
  {#each $locales as lang}
    <a
      href={getLocalizedPath(lang)}
      hreflang={lang}
      aria-current={lang === $locale ? 'page' : undefined}
    >
      {languages[lang]}
    </a>
  {/each}
</nav>
```

### Dropdown Switcher

```svelte
<!-- LanguageDropdown.svelte -->
<script>
  import { page } from '$app/stores';
  import { locale, locales } from '$lib/i18n';

  const languages = {
    en: 'English',
    de: 'Deutsch',
    fr: 'Français',
  };

  let open = $state(false);

  function getLocalizedPath(targetLocale: string) {
    const { pathname } = $page.url;
    const pathWithoutLocale = pathname.replace(/^\/(en|de|fr)/, '') || '/';
    return targetLocale === 'en' ? pathWithoutLocale : `/${targetLocale}${pathWithoutLocale}`;
  }
</script>

<div class="relative">
  <button onclick={() => open = !open}>
    {languages[$locale]}
  </button>

  {#if open}
    <ul class="absolute">
      {#each $locales as lang}
        <li>
          <a
            href={getLocalizedPath(lang)}
            hreflang={lang}
            onclick={() => open = false}
          >
            {languages[lang]}
          </a>
        </li>
      {/each}
    </ul>
  {/if}
</div>
```

## Date & Number Formatting

Use the native `Intl` API for locale-aware formatting.

### Date Formatting

```typescript
// src/lib/i18n/format.ts
import { locale } from '$lib/i18n';
import { get } from 'svelte/store';

export function formatDate(
  date: Date,
  options: Intl.DateTimeFormatOptions = {}
): string {
  const currentLocale = get(locale);

  return new Intl.DateTimeFormat(currentLocale, {
    dateStyle: 'medium',
    ...options,
  }).format(date);
}

export function formatDateTime(date: Date): string {
  const currentLocale = get(locale);

  return new Intl.DateTimeFormat(currentLocale, {
    dateStyle: 'medium',
    timeStyle: 'short',
  }).format(date);
}

export function formatRelative(date: Date): string {
  const currentLocale = get(locale);
  const rtf = new Intl.RelativeTimeFormat(currentLocale, { numeric: 'auto' });

  const diff = date.getTime() - Date.now();
  const days = Math.round(diff / (1000 * 60 * 60 * 24));

  if (Math.abs(days) < 1) {
    const hours = Math.round(diff / (1000 * 60 * 60));
    return rtf.format(hours, 'hour');
  }

  return rtf.format(days, 'day');
}
```

### Number Formatting

```typescript
// src/lib/i18n/format.ts
export function formatNumber(
  value: number,
  options: Intl.NumberFormatOptions = {}
): string {
  const currentLocale = get(locale);

  return new Intl.NumberFormat(currentLocale, options).format(value);
}

export function formatCurrency(value: number, currency = 'USD'): string {
  const currentLocale = get(locale);

  return new Intl.NumberFormat(currentLocale, {
    style: 'currency',
    currency,
  }).format(value);
}

export function formatPercent(value: number): string {
  const currentLocale = get(locale);

  return new Intl.NumberFormat(currentLocale, {
    style: 'percent',
    minimumFractionDigits: 0,
    maximumFractionDigits: 1,
  }).format(value);
}
```

### Usage in Components

```svelte
<script>
  import { formatDate, formatCurrency } from '$lib/i18n/format';

  let { product } = $props();
</script>

<article>
  <h2>{product.name}</h2>
  <p class="price">{formatCurrency(product.price, 'EUR')}</p>
  <p class="date">Added: {formatDate(product.createdAt)}</p>
</article>
```

**Output by locale:**

| Locale | Price | Date |
|--------|-------|------|
| en | €1,234.50 | Jan 15, 2025 |
| de | 1.234,50 € | 15. Jan. 2025 |
| fr | 1 234,50 € | 15 janv. 2025 |

### Alternate Links

Add hreflang tags for search engines:

```svelte
<!-- src/routes/[[lang]]/+layout.svelte -->
<script>
  import { page } from '$app/stores';
  import { locales, locale } from '$lib/i18n';

  const baseUrl = 'https://example.com';

  function getLocalizedPath(targetLocale: string) {
    const { pathname } = $page.url;
    const pathWithoutLocale = pathname.replace(/^\/(en|de|fr)/, '') || '/';
    return targetLocale === 'en' ? pathWithoutLocale : `/${targetLocale}${pathWithoutLocale}`;
  }
</script>

<svelte:head>
  {#each $locales as lang}
    <link
      rel="alternate"
      hreflang={lang}
      href="{baseUrl}{getLocalizedPath(lang)}"
    />
  {/each}
  <link
    rel="alternate"
    hreflang="x-default"
    href="{baseUrl}{$page.url.pathname}"
  />
</svelte:head>
```

## Route-Based Loading

Load translations only for the routes that need them:

```typescript
// src/lib/i18n/index.ts
export const config = {
  loaders: [
    // Common - loaded on all pages
    {
      locale: 'en',
      key: 'common',
      loader: async () => (await import('./en/common.json')).default,
    },
    // Home - only loaded on /
    {
      locale: 'en',
      key: 'home',
      routes: ['/'],
      loader: async () => (await import('./en/home.json')).default,
    },
    // Dashboard - only loaded on /dashboard/*
    {
      locale: 'en',
      key: 'dashboard',
      routes: ['/dashboard'],
      loader: async () => (await import('./en/dashboard.json')).default,
    },
    // Auth - only loaded on /auth/*
    {
      locale: 'en',
      key: 'auth',
      routes: ['/auth/login', '/auth/register'],
      loader: async () => (await import('./en/auth.json')).default,
    },
  ],
};
```

## Content Translation

For user-generated or CMS content stored in the database:

### Schema

```typescript
// src/lib/server/db/schema/content.ts
import { pgTable, text, jsonb } from 'drizzle-orm/pg-core';

export const posts = pgTable('posts', {
  id: text('id').primaryKey(),
  slug: text('slug').notNull(),

  // Translated fields stored as JSON
  title: jsonb('title').$type<Record<string, string>>().notNull(),
  content: jsonb('content').$type<Record<string, string>>().notNull(),

  // Non-translated fields
  authorId: text('author_id').notNull(),
  publishedAt: timestamp('published_at'),
});

// Example data:
// title: { "en": "Hello World", "de": "Hallo Welt" }
```

### Helper Functions

```typescript
// src/lib/i18n/content.ts
import { locale } from '$lib/i18n';
import { get } from 'svelte/store';

export function t<T extends Record<string, string>>(
  translations: T,
  fallback = 'en'
): string {
  const lang = get(locale);
  return translations[lang] ?? translations[fallback] ?? '';
}
```

### Usage

```svelte
<script>
  import { t } from '$lib/i18n/content';

  let { post } = $props();
</script>

<article>
  <h1>{t(post.title)}</h1>
  <div>{@html t(post.content)}</div>
</article>
```

## Alternative: Paraglide

For projects with only 2-5 languages where type safety is the priority:

```typescript
// Paraglide usage (compile-time)
import * as m from '$lib/paraglide/messages';

m.greeting({ name: 'World' }); // Fully typed
```

Trade-offs:
- (+) Full type safety
- (+) Smaller bundle for few languages
- (-) Bundles all languages together
- (-) No runtime language switching

## Summary

| What | How |
|------|-----|
| Library | sveltekit-i18n |
| Messages | `src/lib/i18n/{locale}/*.json` |
| Usage | `$t('key.path', { params })` |
| Routing | `[[lang]]` optional parameter |
| Per-route loading | `routes` array in loaders |
| Dates/Numbers | `Intl` API |
| DB content | JSON columns with helper |
